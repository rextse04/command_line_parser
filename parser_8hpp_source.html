425 std::array<bool, Info.tree.size()> visited{};

429 *front = node_loc;

432 *(back++) = node_idx;
433 visited[node_idx] = true;
434 }

439 case option:
440 case variable_option: {

444 *front = next_option;

450 push(node.next);

467 constexpr auto parse(Args&& args) noexcept {
468 using enum parse_node_type;
469 using enum error_type;
470 constexpr bool full_result = std::ranges::forward_range<Args>;
471 using return_type = std::conditional_t<full_result,
472 std::expected<parse_result<Args>, parse_error<Args>>,
473 std::expected<part_parse_result, part_parse_error>>;
474 auto node = Info.tree.begin();
475 auto start_node = node;
476 auto arg_current = ranges::begin(args);
477 auto arg_end = ranges::end(args);
478 std::size_t arg_loc = 0;
479 auto next_arg = [&start_node, &node, &arg_current, &arg_loc](std::size_t next_node_idx) {
480 start_node = node = Info.tree.begin() + next_node_idx;
481 ++arg_current; ++arg_loc;
482 };
483 auto raise = [&start_node, &args, &arg_loc](error_type err, std::size_t in_arg_loc = 0) {
484 refs_type refs = search_refs(start_node - Info.tree.begin());
485 if constexpr (full_result) {
486 return return_type{std::unexpect, err, std::forward<Args>(args), error_loc{arg_loc, in_arg_loc}, refs};
487 } else {
488 return return_type{std::unexpect, err, refs};
489 }
490 };
491 while (true) {
492 if (arg_current == arg_end) [[unlikely]] {
493 while (node->next_placeholder) {
494 node = Info.tree.begin() + node->next_placeholder;
495 }
496 if (node->type != end) {
497 return raise(too_few_arguments);
498 }
499 }
500 switch (node->type) {
501 case option:
502 case variable_option:
503 if (node->option_name == *arg_current) {
504 if (node->type == variable_option) {
505 vars_[node->var_index] = {string_type{*arg_current}, {arg_loc, 0}};
506 }
507 next_arg(node->next);
508 } else if (node->next_placeholder) {
509 node = Info.tree.begin() + node->next_placeholder;
510 } else [[unlikely]] {
511 return raise(unknown_option);
512 }
513 continue;
514 case variable:
515 if ((*arg_current).starts_with(config.specials.flag_prefix)) {
516 return raise(flag_cannot_be_variable);
517 }
518 vars_[node->var_index] = {string_type{*arg_current}, {arg_loc, 0}};
519 next_arg(node->next);
520 continue;
521 case end:
522 break;
523 }
524 break;
525 }
526 for (;arg_current != arg_end; ++arg_current) {
527 std::basic_string_view<char_type> flag_str = *arg_current;
528 if (flag_str.starts_with(config.specials.flag_prefix)) [[likely]] {
529 const std::size_t eq_pos = flag_str.find(config.specials.equal);
530 std::basic_string_view<char_type> flag_name = flag_str.substr(0, eq_pos);
531 const std::size_t h = get_hash<hash_type, Info.flag_set.size()>(flag_name);
532 const auto& flag = Info.flag_set[h];
533 if (flag.name == flag_name && flag.defined_for[node->usage_index]) [[likely]] {
534 flags_[h] = true;
535 if (eq_pos != flag_str.npos) {
536 if (std::size_t var_index = flag.var_index_for[node->usage_index]) [[likely]] {
537 vars_[var_index] = {
538 string_type{flag_str.substr(eq_pos + config.specials.equal.size())},
539 {arg_loc, eq_pos + 1}

541 } else {
542 return raise(flag_does_not_accept_argument, eq_pos);
543 }
544 }
545 } else {
546 return raise(unknown_flag);
547 }
548 } else {
549 return raise(too_many_arguments);
550 }
551 }
552 if constexpr (full_result) {
553 return return_type{std::in_place,
554 Info.usages[node->usage_index].name, node->usage_index, std::forward<Args>(args)};
555 } else {
556 return return_type{std::in_place, Info.usages[node->usage_index].name, node->usage_index};

674 class usage_parse_error {

677 using char_type = config_type::super_type::char_type;
678 using string_view_type = std::basic_string_view<char_type>;
679 static constexpr bool ref_enabled = std::is_same_v<char_type, char>;
680 private:
681 static constexpr std::string_view intro = "Parse error.\n \032 | \032\n \032 | \032\n\032";
682 static constexpr std::string_view ref_fallback = "Preview is not available if char_type is not char.";
683 static constexpr std::size_t ref_size = []() -> std::size_t {
684 if constexpr (ref_enabled) {
685 return ranges::max(Config.usages | views::transform([](const auto& usage) {
686 return usage.format.size();
687 }));
688 } else {
689 return ref_fallback.size();
690 }
691 }();
692 static constexpr std::size_t num_max_size = sizeof(std::size_t) * CHAR_BIT / 3; // overestimation
693 public:
694 static constexpr std::size_t what_size =
695 intro.size() + (num_max_size + ref_size) * 2 + usage_parse_msg_max_size; // +'\0'
696 private:
697 std::array<char, what_size> _what{}; // null-terminated
698 public:
699 constexpr usage_parse_error(std::string_view msg, std::size_t usage_idx, std::size_t loc) noexcept {
700 auto sec_start = intro.begin(), sec_end = sec_start - 1;
701 char* current = _what.data();
702 auto copy_sec = [&sec_start, &sec_end, &current]() {
703 sec_start = sec_end + 1;
704 sec_end = ranges::find(sec_start, intro.end(), '\032');
705 current = ranges::copy(sec_start, sec_end, current).out;
706 };
707 copy_sec();
708 char* after_num = std::to_chars(current, &*_what.end(), usage_idx + 1).ptr;
709 std::size_t num_size = after_num - current;
710 current = after_num;
711 copy_sec();
712 if constexpr (ref_enabled) {
713 current = ranges::copy(Config.usages[usage_idx].format, current).out;
714 } else {
715 current = ranges::copy(ref_fallback, current).out;
716 }
717 copy_sec();
718 current = ranges::fill(current, current + num_size, ' ');
719 copy_sec();
720 current = ranges::fill(current, current + loc, ' ');
721 *(current++) = '^';
722 copy_sec();
723 ranges::copy(msg, current);
724 }
