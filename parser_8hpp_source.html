436 auto push = [&back, &visited](std::size_t node_idx) {

474 using enum parse_node_type;
475 using enum error_type;
476 using args_type = std::remove_cvref_t<Args>;
477 using iter_type = ranges::iterator_t<args_type>;
478 using sentinel_type = ranges::sentinel_t<args_type>;
479 constexpr bool full_result = ranges::forward_range<Args>;
480 using return_type = std::conditional_t<full_result,
481 std::expected<parse_result<Args>, parse_error<Args>>,
482 std::expected<part_parse_result, part_parse_error>>;
483 auto node = Info.tree.begin();
484 auto start_node = node;
485 auto arg_current = ranges::begin(args);
486 auto arg_end = ranges::end(args);
487 std::size_t arg_loc = 0;
488 auto next_arg = [&arg_current, &arg_loc]() {
489 ++arg_current; ++arg_loc;
490 };
491 auto next_arg_node = [&start_node, &node, next_arg](std::size_t next_node_idx) {
492 start_node = node = Info.tree.begin() + next_node_idx;
493 next_arg();
494 };
495 auto raise = [&start_node, &args, &arg_loc](error_type err, std::size_t in_arg_loc = 0) {
496 refs_type refs = search_refs(start_node - Info.tree.begin());
497 if constexpr (full_result) {
498 return return_type{std::unexpect,
499 err, std::forward<Args>(args), error_loc{arg_loc, in_arg_loc}, refs};
500 } else {
501 return return_type{std::unexpect, err, refs};
502 }
503 };
504 string_view_type arg;
505 while (true) {
506 if (arg_current == arg_end) [[unlikely]] {
507 while (node->next_placeholder) {
508 node = Info.tree.begin() + node->next_placeholder;
509 }
510 if (node->type != variadic && node->type != end) {
511 return raise(too_few_arguments);
512 }
513 } else {
514 arg = *arg_current;
515 }
516 parse_arg:
517 switch (node->type) {
518 case option:
519 case variable_option: {
520 if (node->option_name == arg) {
521 if (node->type == variable_option) {
522 vars_[node->var_index] = {string_type{arg}, {arg_loc, 0}};
523 }
524 next_arg_node(node->next);
525 } else if (node->next_placeholder) {
526 node = Info.tree.begin() + node->next_placeholder;
527 if !consteval {goto parse_arg;}
528 } else [[unlikely]] {
529 return raise(unknown_option);
530 }
531 continue;
532 }
533 case variable: {
534 if (arg.starts_with(config.specials.flag_prefix)) {
535 return raise(flag_cannot_be_variable);
536 }
537 vars_[node->var_index] = {string_type{arg}, {arg_loc, 0}};
538 next_arg_node(node->next);
539 continue;
540 }
541 case variadic: {
542 if constexpr (std::sized_sentinel_for<iter_type, sentinel_type>) {
543 variadic_.reserve(ranges::end(args) - arg_current);
544 } else if constexpr (ranges::sized_range<args_type>) {
545 variadic_.reserve(ranges::size(args));
546 }
547 for (; arg_current != arg_end; next_arg()) {
548 arg = *arg_current;
549 if (arg.starts_with(config.specials.flag_prefix)) break;
550 variadic_.emplace_back(arg);
551 }
552 break;
553 }
554 case end: {
555 break;
556 }
557 }
558 break;
559 }
560 for (;arg_current != arg_end; ++arg_current) {
561 string_view_type flag_str = *arg_current;
562 if (flag_str.starts_with(config.specials.flag_prefix)) [[likely]] {
563 const std::size_t eq_pos = flag_str.find(config.specials.equal);
564 string_view_type flag_name = flag_str.substr(0, eq_pos);
565 const std::size_t h = get_hash<hash_type, Info.flag_set.size()>(flag_name);
566 const auto& flag = Info.flag_set[h];

588 Info.usages[node->usage_index].name, node->usage_index, std::forward<Args>(args)};
